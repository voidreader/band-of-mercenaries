### 2.2 weekly_contributions 테이블 CREATE

페이즈 1 #6 §3.5.1 스키마 그대로:

```sql
CREATE TABLE IF NOT EXISTS public.weekly_contributions (
  user_id UUID NOT NULL REFERENCES auth.users(id) ON DELETE CASCADE,
  week_start_utc DATE NOT NULL,
  stabilization_score INT NOT NULL DEFAULT 0 CHECK (stabilization_score >= 0),
  faction_score INT NOT NULL DEFAULT 0 CHECK (faction_score >= 0),
  combat_score INT NOT NULL DEFAULT 0 CHECK (combat_score >= 0),
  crafting_score INT NOT NULL DEFAULT 0 CHECK (crafting_score >= 0),
  achievement_score INT NOT NULL DEFAULT 0 CHECK (achievement_score >= 0),
  total_score INT GENERATED ALWAYS AS (
    stabilization_score
    + faction_score
    + combat_score
    + crafting_score
    + achievement_score
  ) STORED,
  updated_at TIMESTAMPTZ NOT NULL DEFAULT now(),
  PRIMARY KEY (user_id, week_start_utc)
);

COMMENT ON TABLE public.weekly_contributions IS
  'M8.5 주간 기여도 랭킹: 사용자×주간 점수 기록. UTC 월요일 0:00 사이클.';
COMMENT ON COLUMN public.weekly_contributions.user_id IS
  'Supabase 익명 인증 user ID (auth.users 참조)';
COMMENT ON COLUMN public.weekly_contributions.week_start_utc IS
  'UTC 월요일 0:00의 날짜 (주간 경계)';
COMMENT ON COLUMN public.weekly_contributions.total_score IS
  '5항목 자동 합산 (GENERATED ALWAYS AS STORED). 클라이언트는 별도 전송하지 않음.';
```

### 2.3 인덱스

```sql
-- top 100 lookup 최적화 (week_start_utc + total_score DESC)
CREATE INDEX IF NOT EXISTS idx_weekly_contributions_total_score
  ON public.weekly_contributions(week_start_utc, total_score DESC);

-- 자기 위치 lookup 최적화 (week_start_utc + user_id, PK로 충분하지만 명시)

```

### 2.4 RLS 정책 4종

페이즈 1 #6 §3.5.3 그대로:

```sql
-- Row Level Security 활성화
ALTER TABLE public.weekly_contributions ENABLE ROW LEVEL SECURITY;

-- 본인만 INSERT
CREATE POLICY "Insert own row"
  ON public.weekly_contributions
  FOR INSERT
  WITH CHECK (auth.uid() = user_id);

-- 본인만 UPDATE
CREATE POLICY "Update own row"
  ON public.weekly_contributions
  FOR UPDATE
  USING (auth.uid() = user_id)
  WITH CHECK (auth.uid() = user_id);

-- 모든 인증 사용자 SELECT (top 100 lookup)
-- anon key + 미인증 상태에서도 SELECT 가능하도록 USING (true)
CREATE POLICY "Select all"
  ON public.weekly_contributions
  FOR SELECT
  USING (true);

-- DELETE는 RLS 정책 없음 = 일반 사용자 거부
-- pg_cron 시스템 권한(SECURITY DEFINER 또는 cron 전용 role)만 삭제 가능
```

### 2.5 pg_cron 정리 정책

페이즈 1 #6 §3.5.4 — 매주 수요일 UTC 0:00 자동 삭제:

SELECT cron.schedule(
  'cleanup_old_weekly_contributions',

### 2.6 RPC 함수 `upsert_weekly_score`

페이즈 1 #6 §3.5.5 그대로 (MVP 채택):

```sql
CREATE OR REPLACE FUNCTION public.upsert_weekly_score(
  p_stab INT,
  p_fac INT,
  p_com INT,
  p_cra INT,
  p_ach INT
) RETURNS void
LANGUAGE plpgsql
SECURITY DEFINER
SET search_path = public
AS $$
DECLARE
  v_week_start DATE := date_trunc('week', timezone('UTC', now()))::date;
BEGIN
  -- 인증 가드
  IF auth.uid() IS NULL THEN
    RAISE EXCEPTION 'authentication required';
  END IF;

  -- 음수 점수 거부 (이상치 탐지 1차)
  IF p_stab < 0 OR p_fac < 0 OR p_com < 0 OR p_cra < 0 OR p_ach < 0 THEN
    RAISE EXCEPTION 'weekly contribution scores must be non-negative';
  END IF;

  -- UPSERT (PK 충돌 시 갱신)
  INSERT INTO public.weekly_contributions (
    user_id,
    week_start_utc,
    stabilization_score,
    faction_score,
    combat_score,
    crafting_score,
    achievement_score
  ) VALUES (
    auth.uid(),
    v_week_start,
    p_stab, p_fac, p_com, p_cra, p_ach
  )
  ON CONFLICT (user_id, week_start_utc) DO UPDATE SET
    stabilization_score = EXCLUDED.stabilization_score,
    faction_score = EXCLUDED.faction_score,
    combat_score = EXCLUDED.combat_score,
    crafting_score = EXCLUDED.crafting_score,
    achievement_score = EXCLUDED.achievement_score,
    updated_at = now();
END;
$$;

-- 인증 사용자만 호출 가능
GRANT EXECUTE ON FUNCTION public.upsert_weekly_score(INT, INT, INT, INT, INT)
  TO authenticated;

-- anon role은 명시적으로 거부 (미인증 클라이언트 호출 차단)
REVOKE EXECUTE ON FUNCTION public.upsert_weekly_score(INT, INT, INT, INT, INT)
  FROM anon;
```

### 2.7 (선택) 자기 위치 lookup RPC

페이즈 2 #5 §6.2에서 권장한 동률 처리 정합 RPC. 클라이언트가 직접 쿼리해도 되지만 RPC로 캡슐화:

```sql
CREATE OR REPLACE FUNCTION public.get_my_weekly_rank()
RETURNS TABLE(
  my_rank BIGINT,
  total_users BIGINT,
  my_total_score INT
)
LANGUAGE plpgsql
SECURITY DEFINER
SET search_path = public
AS $$
DECLARE
  v_week_start DATE := date_trunc('week', timezone('UTC', now()))::date;
  v_my_score INT;
  v_my_updated TIMESTAMPTZ;
BEGIN
  IF auth.uid() IS NULL THEN
    RAISE EXCEPTION 'authentication required';
  END IF;

  -- 자기 점수 lookup
  SELECT total_score, updated_at
    INTO v_my_score, v_my_updated
    FROM public.weekly_contributions
    WHERE user_id = auth.uid()
      AND week_start_utc = v_week_start;

  IF v_my_score IS NULL THEN
    -- 이번 주 점수 0 (아직 미등록)
    RETURN QUERY
      SELECT
        (SELECT COUNT(*) FROM public.weekly_contributions WHERE week_start_utc = v_week_start) + 1,
        (SELECT COUNT(*) FROM public.weekly_contributions WHERE week_start_utc = v_week_start),
        0;
    RETURN;
  END IF;

  -- 자기 위치 산출 (동률 시 페이즈 2 #5 §6.2 정책)
  RETURN QUERY
    SELECT
      (SELECT COUNT(*) + 1
       FROM public.weekly_contributions
       WHERE week_start_utc = v_week_start
         AND (total_score > v_my_score
              OR (total_score = v_my_score AND updated_at < v_my_updated)
              OR (total_score = v_my_score AND updated_at = v_my_updated
                  AND user_id < auth.uid()))),
      (SELECT COUNT(*) FROM public.weekly_contributions WHERE week_start_utc = v_week_start),
      v_my_score;
END;
$$;

GRANT EXECUTE ON FUNCTION public.get_my_weekly_rank() TO authenticated;
REVOKE EXECUTE ON FUNCTION public.get_my_weekly_rank() FROM anon;
```

SELECT table_name
  FROM information_schema.tables
  WHERE table_schema = 'public' AND table_name = 'weekly_contributions';
-- 기대: 1 row

-- 2. 인덱스 확인
SELECT indexname FROM pg_indexes
  WHERE tablename = 'weekly_contributions';
-- 기대: weekly_contributions_pkey + idx_weekly_contributions_total_score

-- 3. RLS 정책 확인
SELECT policyname, cmd
  FROM pg_policies
  WHERE tablename = 'weekly_contributions';
-- 기대: 3 rows (Insert own row / Update own row / Select all)

-- 4. pg_cron job 확인
SELECT jobname, schedule, command
  FROM cron.job
  WHERE jobname = 'cleanup_old_weekly_contributions';
-- 기대: 1 row, schedule = '0 0 * * 3'

-- 5. RPC 함수 확인
SELECT routine_name, security_type

  WHERE routine_schema = 'public'

## 5. RLS 정책 검증 (수동 테스트)

페이즈 4 #6 spec 구현 후 다음을 확인:

| 테스트 | 기대 결과 |
|--------|---------|
| 익명 인증 user A → INSERT 자기 행 | ✅ 성공 |
| 익명 인증 user A → INSERT user B 행 (user_id 위조) | ❌ RLS 차단 |
| 익명 인증 user A → UPDATE 자기 행 | ✅ 성공 |
| 익명 인증 user A → UPDATE user B 행 | ❌ RLS 차단 |
| 익명 인증 user A → SELECT 전체 (top 100) | ✅ 성공 |
| 익명 인증 user A → DELETE 자기 행 | ❌ RLS 정책 없음 = 차단 |
| 미인증 anon → SELECT 전체 | ✅ 성공 (top 10 미리보기) |
| 미인증 anon → upsert_weekly_score RPC | ❌ REVOKE EXECUTE FROM anon |
| 인증 user → upsert_weekly_score RPC | ✅ SECURITY DEFINER + auth.uid() 가드 |
| 인증 user → upsert_weekly_score 음수 점수 | ❌ RAISE EXCEPTION |

## 7. 운영 노트

### 7.1 점수 위조 모니터링

페이즈 2 #5 §1.2 + 페이즈 1 #6 §3.6 — MVP는 클라이언트 신뢰. 다만 운영 도구에서 이상치 탐지:

```sql
-- 1주에 이상치 검출 (베테랑 cap 합 ~14500 기준 30000 이상)
SELECT user_id, week_start_utc, total_score
  FROM public.weekly_contributions
  WHERE total_score > 30000
  ORDER BY total_score DESC LIMIT 20;
```

operation-bom에 별도 모니터링 페이지 작업 위임.

### 7.2 pg_cron 권한 검증

Supabase Pro+ tier 미사용 시 pg_cron 확장이 불가능. 무료 tier에서 동등 기능 구현:
- 옵션 A: Supabase Edge Function + 외부 cron (예: GitHub Actions schedule)
- 옵션 B: 클라이언트 측에서 주간 경계 감지 시 만료 행 DELETE 호출 (RLS 우회 필요 → 부적합)

본 산출물은 Pro+ tier 전제. 페이즈 4 #6 spec에서 환경 변수 검증.

### 7.3 데이터 마이그레이션 패턴

본 테이블은 사용자 데이터 (vs 정적 데이터). `SyncService.allTables` 등록 안 함. `data_versions` 갱신 안 함. Flutter 앱은 `Supabase.instance.client.rpc('upsert_weekly_score', ...)` 또는 직접 쿼리로 접근.

### 7.4 익명 인증 활성화 절차

Supabase Dashboard → Authentication → Providers → Anonymous Sign-in **Enabled** 설정 필요. 페이즈 4 #6 spec에 운영 작업 명시.
